347 double Management::haversine(double lat1, double lon1, double lat2, double lon2) {
348  const double R = 6371.0; // Raio médio da Terra em quilômetros
349 
350  lat1 = toRadians(lat1);
351  lon1 = toRadians(lon1);
352  lat2 = toRadians(lat2);
353  lon2 = toRadians(lon2);
354 
355  double dlat = lat2 - lat1;
356  double dlon = lon2 - lon1;
357 
358  double a = sin(dlat / 2.0) * sin(dlat / 2.0) + cos(lat1) * cos(lat2) * sin(dlon / 2.0) * sin(dlon / 2.0);
359  double c = 2.0 * atan2(sqrt(a), sqrt(1.0 - a));
360 
361  double distance = R * c;
362 
363  return distance;
364 }
